 int fib(int n)
 {
   if(n<=1)
     return n;
   return(fib(n-1)+fib(n-2));
 }

  {
    for(s=num-i; s>=1; s--)
       printf(" ");
    printf("%d", i);
    for(s=i*2; s>1; s--)
       printf(" ");
    printf("%d", i);
    printf("\n");
  }
  for(i=1,n=num-1; i<num; i++,n--)
  {
    for(s=i; s>=1; s--)
       printf(" ");
    printf("%d",n);
    for(s=n*2; s>1; s--)
       printf(" ");
    printf("%d", n);
    printf("\n");
