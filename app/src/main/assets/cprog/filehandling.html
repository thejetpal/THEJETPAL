Know the last date of modification of any file 


#include <stdio.h>
#include <time.h>
#include <sys\stat.h>
int main(){
    struct stat s;
    FILE *fp;
    fp=fopen("exmple.txt","r");
    fstat(fileno(fp),&s);
    printf("Last date of modification : %s",ctime(&s.st_ctime));
    return 0;
}

Count total number of character and word in a file 


#include<stdio.h>
#include<stdlib.h>
void main()
{
    FILE *fp;
    char ch;
    int tc=0,tw=1;
    fp=fopen("file.txt","r");
    if(fp==NULL)
    {
    	printf("File doesn't exist\n");
    	exit(1);
    }
    while(!feof(fp))
    {
    	ch=getc(fp);
    	tc++;
    	if(ch==' ')
    	{
    		tw++;
    	}
    }
    printf("Total character: %d \nTotal words: %d",tc,tw);

}
